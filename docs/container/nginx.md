# Nginx
- 代理服务器
## install
- brew install nginx
- /usr/local/etc/nginx/nginx.conf
- /usr/local/etc/nginx/logs
## command
- nginx --start nginx
- nginx -s stop --fast shutdown
- nginx -s reload --reload the configuration file
- nginx -s quit --graceful shutdown
- nginx -s reopen --reopening the log files
